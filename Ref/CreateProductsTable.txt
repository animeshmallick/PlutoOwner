CREATE TABLE `shop`.`products` (`product_id` INT(5) NOT NULL AUTO_INCREMENT , `product_tags` VARCHAR(512) NOT NULL , `product_name` VARCHAR(30) NOT NULL , `product_description` TEXT NOT NULL , `product_unit` VARCHAR(20) NOT NULL , `product_unit_mrp` DOUBLE NOT NULL , `product_unit_discount` FLOAT NOT NULL , `product_keywords` TEXT NOT NULL , `product_unit_cost` DOUBLE NOT NULL , `product_image_key` VARCHAR(30) NOT NULL , `product_is_restricted` TINYINT NOT NULL , `product_inventory` INT(11) NOT NULL , `product_last_updated` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP , `product_listed_at` DATETIME NOT NULL , `product_is_buyable` TINYINT NOT NULL DEFAULT '1' , PRIMARY KEY (`product_id`)) ENGINE = InnoDB;
